CREATE TABLE futebolize_dw (
	id INT AUTO_INCREMENT NOT NULL PRIMARY KEY,
	id_field INT NOT NULL,
	id_costumer INT NOT NULL,
	id_owner INT NOT NULL,
	belongs_to INT NOT NULL
)

CREATE TABLE fields (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(300) NOT NULL,
    score INT,
    address VARCHAR(300) NOT NULL,
    state VARCHAR(100) NOT NULL,
    max_players INT,
    deleted_at DATETIME DEFAULT NULL,
    checksum VARBINARY(64) GENERATED ALWAYS AS (SHA2(CONCAT(id, address, state), 256)) STORED,
    CONSTRAINT check_score CHECK (score >= 1 AND score <= 5)
);

CREATE TABLE available_schedules (
	id_field INT,
	schedules DATETIME,
	is_available TINYINT(1) NOT NULL DEFAULT 0
)

CREATE TABLE owner (
	id INT AUTO_INCREMENT NOT NULL,
	name VARCHAR(300) NOT NULL,
	cnpj INT NOT NULL,
	phone_number INT NOT NULL,
	address VARCHAR(300) NOT NULL
)

CREATE TABLE costumer (
	id INT AUTO_INCREMENT NOT NULL,
	name VARCHAR(300) NOT NULL,
	cpf INT NOT NULL,
	address VARCHAR(300) NOT NULL,
	phone_number INT NOT NULL,
	email VARCHAR(100)
)
